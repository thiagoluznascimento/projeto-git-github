git init 					inicializa um novo repositório

git pull					baixa a versão mais atual para sua area de trabalho

git add .					adiciona os arquivos atuais ao próxio commit

git status					verifica o status atual do repositório git

git commit -m "mensagem do comit"	cria um commit com uma mensagem 

git push					envia atualizaçãoes para a nuvem na branch atualmente ativa

git branch					permite listar e ver qual branch está ativa atualmente

git checkout -b "nome da branch"	permite mudar para uma nova branch com base em outra

git merge "branch a receber merge"	permite fazer o merge da branch ativa atualmente com outra branch

git pull					atualiza a branch atualmente ativa

touch .gitignore				iginora os aquivos que vc não quer que suba para o servidor git (tem que add os aquivos no novo arquivo criado automaticamente)
